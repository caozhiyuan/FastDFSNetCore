## Ngx_fastdfs_module


``` nginx
server {
          listen       80;
 
          location / {
            root   /path/to/fdfsstore/fdfs_storage/data;
          }
 
          location /group1/M00 {
            root /path/to/fdfsstore/fdfs_storage/data;
            ngx_fastdfs_module;                   
         }  
          
         location ~* /group1/M00/(.*?)_([0-9]+)X([0-9]+)\.(jpeg|jpg|gif|png)$ {
           root /path/to/fdfsstore/fdfs_storage/data;       
           set $h $2;
           set $w $3;
           rewrite /group1/M00/(.*?)_([0-9]+)X([0-9]+)\.(jpeg|jpg|gif|png)$ /$1.$4 break;
           image_filter_buffer 10M;
           image_filter resize $h $w;                 
        }        
}
```
